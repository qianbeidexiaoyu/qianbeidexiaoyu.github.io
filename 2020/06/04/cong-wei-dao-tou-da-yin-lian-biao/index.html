从尾到头打印链表

剑指offer 单链表

题目链接

LeetCode

题目描述

输入一个链表的头节点，从尾到头反过来返回每个节点的值（用数组返回）

示例

输入：head = [1,3,2]
输出：[2,3,1]`

限制

0 <= 链表长度 <= 10000

解题思路

  1. 递归
  2. 用栈来存储遍历序列
  3. 再构造一个链表并输出

错误总结

  1. 数组传递引用必须指定数组大小，函数形参形式 eg: (&arr)[3]，不推荐使用数组的引用作为函数参数,因为要给定数组大小，因此在传递数组时一般使用指针，但使用指针无法通过(sizeof(arr) / sizeof(arr[0])获取数组的元素个数，因此一般在函数形参列表中再给定义一个参数eg: LinkList array2linklist(ElemType* arr,int n)

    代码部分

    使用递归算法：单链表结构包含头结点

    #include<iostream>
#include<vector>
using namespace std;
//单链表基本结构
#define ElemType int
typedef struct LNode {
 ElemType data;
 struct LNode* next;
}LNode, *LinkList;
//传入一个链表指针，生成一个带有空的头结点的单链表
bool InitList(LinkList& L) {
 LNode* head = new(LNode);
 if (!head)
     return false;
 L = head;
 head->next = nullptr;
 head->data = NULL;
 return true;
}
//求表长
int Length(LinkList L) {
 int num_node = 0;
 LNode* p = L;
 while (p->next) {
     p = p->next;
     num_node++;
 }
 return num_node;
}
//在单链表L上的第i个位置插入data=e的节点 头结点为第0个节点
bool ListInsert(LinkList& L, int i, ElemType e) {
 //新建节点
 LNode* new_node = new(LNode);
 if (!new_node)
     return false;
 new_node->next = nullptr;
 new_node->data = e;
 //插入的位置区间为[1,Length+1]
 if (i<1 || i > Length(L) + 1)
     return false;
 //找到第i-1个节点
 LNode* p = L;
 //p走到第i-1个位置，并将新节点插入到第i个位置
 for (int j = 0; j < i - 1; j++)
     p = p->next;
 //将新节点插入到p之后
 new_node->next = p->next;
 p->next = new_node;
 return true;
}
//输出操作。按前后顺序输出线性表L的所有元素值
void PrintList(LinkList L) {
 LNode* p = L;
 if (!p->next) {
     cout << "链表为空" << endl;
     return;
 }
 while (p->next) {
     p = p->next;
     cout << p->data << " ";
 }
 cout << endl;
}
//输入一个数组，将其存储成单链表的形式
//@n:数组元素的个数
LinkList array2linklist(ElemType* arr,int n) {
 LinkList L;
 InitList(L);
 for (int i = 1; i < n + 1; i++) {
     ListInsert(L, i, arr[i - 1]);
 }
 return L;
}
//反转输出单链表，返回一个vector<int>
vector<int> reversePrint(LNode* head){
 vector<int> out;
 if (head){
     out = reversePrint(head->next);
     out.push_back(head->data);
 }
 return out;
}
//主函数测试
int main() {
 ElemType arr[3] = { 1,3,2 };
 int n = sizeof(arr) / sizeof(arr[0]);
 vector<int> out;
 LinkList head = array2linklist(arr,n);
 PrintList(head);
     out = reversePrint(head->next); //带有一个头结点，LeetCode没有，所以传值直接传head->next
 //输出打印
 for (int i = 0; i < out.size(); i++) {
     cout << out[i] << " ";
 }
 cout << endl;
 system("pause");
 return 0;
}
