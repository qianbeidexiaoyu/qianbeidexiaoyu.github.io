线性表的基本操作

InitList(&L)：初始化表，构造一个空的线性表。
Length(L):求表长。返回线性表L的长度，即L中数据元素的个数。
LocateElem(L,e):按值查找操作。在表L中查找具有给定关键字值的元素。
GetElem(L,i)：按位查找操作。获取表L中第i个位置的元素的值。
ListInsert(&L,i,e):插入操作。在表L中第i个位置上插入指定元素e。
ListDelete(&L,i,&e):删除操作。删除表L中第i个位置的元素，并用e返回删除元素的值。
PrintList(L):输出操作。按前后顺序输出线性表L的所有元素值。
Empty(L):判空操作，若L为空表，则返回true，否则返回false，
Destroylist(&L):销毁操作。销毁线性表，并释放线性表L所占用的内存空间。

线性表的顺序存储-顺序表:相关操作基于动态分配

宏定义

# define ElemType int
# define MaxSize 100            //线性表最大长度
# define SeqList_INITSIZE 100    //表长度初始定义
# define SeqList_INCREMENT 10    //当顺序表满，动态扩充顺序表的增量

结构体定义

//结构体定义-静态分配
typedef struct {
    ElemType data[MaxSize];
    int length;
}SeqList;

//结构体定义-动态分配
typedef struct{
    ElemType* data;
    int length;
    int MaxSize;
}SeqList;

初始化

//初始化表，构造一个空的线性表。
bool InitList(SeqList& L){
    L.data = new ElemType[SeqList_INITSIZE];
    if (L.data == NULL)
        return false;
    L.length = 0;
    L.MaxSize = SeqList_INITSIZE;
    return true;
}

求表长

int Length(SeqList L) {
    return L.length;
}

按值查找

int LocateElem(SeqList L, ElemType e) {
    for (int i = 0; i < L.length; i++) {
        if (L.data[i] == e)
            return i;
    }
    return -1;
}

按索引查找

ElemType GetElem(SeqList L,int i) {
    return L.data[i];
}

插入

bool ListInsert(SeqList& L,int i,ElemType e)
{
    ElemType *data_new, *p, *q;
    if (i<1 || i>L.length + 1) 
        return false;
    if (L.length >= L.MaxSize){
        data_new = new ElemType[L.MaxSize+ SeqList_INCREMENT];
        if (!data_new)
            return false;
        L.data = data_new;
        L.MaxSize += SeqList_INCREMENT;
    }
    q = &(L.data[i - 1]);
    for (p = &(L.data[L.length - 1]); p >= q; p--)
        *(p + 1) = *p;
    *q = e;
    L.length++;
    return true;
}

删除

bool ListDelete(SeqList& L,int i,ElemType& e){
    ElemType *p, *q;
    if (i<1 || i>L.length)
        return false;
    p = &(L.data[i - 1]);
    e = *p;
    q = L.data + L.length - 1;
    for (++p; p <= q; ++p)
        *(p - 1) = *p;
    L.length--;
    return true;
}

打印

bool PrintList(SeqList L){
    int i;
    if (L.length<1)
        return false;
    else{
        for (i = 0; i<L.length; i++)
            cout << L.data[i] << "   ";
    }
    return true;
}

判空

bool Empty(SeqList L) {
    if (L.Length == 0)
        return true;
    return false;
}

销毁

void Destroylist(SeqList& L) {
    if (L.data)
        free(L.data);
}

线性表的链式存储-单链表

宏定义

# define ElemType int

结构体定义

typedef struct LNode{
    ElemType data;
    struct LNonde* next;
}LNode,*LinkList;

初始化链表

//传入一个链表指针，生成一个带有空的头结点的单链表
bool InitList(LinkList& L){
    LNode* head = new(LNode);
    if (!head)
        return false;
    L = head;
    head->next = nullptr;
    head->data = NULL;
    return true;
}

求表长

int Length(LinkList L) {
    int num_node = 0;
    LNode* p = L;
    while (p->next) {
        p = p->next;
        num_node++;
    }
    return num_node;
}

按值查找

int LocateElem(LinkList L, ElemType e) {
    LNode* p = L;
    int index = 0;
    while (p->next) {
        p = p->next;
        index++;
        if (p->data == e)
            return index;
    }
    return -1;
}

按索引查找

ElemType GetElem(LinkList L, int i) {
    if (i<1 || i > Length(L))
        return NULL;
    LNode* p = L;
    for (int j = 0; j < i; j++)
        p = p->next;
    return p->data;
}

插入

//在单链表L上的第i个位置插入data=e的节点 头结点为第0个节点
bool ListInsert(LinkList& L,int i, ElemType e){
    //新建节点
    LNode* new_node = new(LNode);
    if (!new_node)
        return false;
    new_node->next = nullptr;
    new_node->data = e;
    //插入的位置区间为[1,Length+1]
    if (i<1 || i > Length(L) +1)
        return false; 
    //找到第i-1个节点
    LNode* p = L;
    //p走到第i-1个位置，并将新节点插入到第i个位置
    for (int j = 0; j < i-1; j++)
        p = p->next;
    //将新节点插入到p之后
    new_node->next = p->next;
    p->next = new_node;
    return true;
}

删除

//删除操作。删除表L中第i个位置的元素，并用e返回删除元素的值。
bool ListDelete(LinkList& L,int i, ElemType& e) {
    if (i<1 || i > Length(L))
        return false;
    //找到第i个节点
    LNode* p = L;
    LNode* pre = p;
    for (int j = 0; j < i; j++) {
        pre = p;
        p = p->next;
    }
    //保存被删除的值
    e = p->data;
    pre->next = p->next;
    delete(p);
    p = nullptr;
    return true;
}

打印

//按前后顺序输出线性表L的所有元素值
void PrintList(LinkList L) {
    LNode* p = L;
    if (!p->next){
        cout << "链表为空" << endl;
        return;
    }
    while(p->next) {
        p = p->next;
        cout << p->data << " ";
    }
    cout << endl;
}

判空

//若L为空表，则返回true，否则返回false
bool Empty(LinkList L) {
    LNode* p = L;
    if (p->next)
        return true;
    else
        return false;
}

销毁

//销毁操作。销毁线性表，并释放线性表L所占用的内存空间,包括头结点
bool Destroylist(LinkList &L) {
    LNode* p = L;
    LNode* delete_ptr = p;
    while (p->next) {
        delete_ptr = p;
        p = p->next;
        delete(delete_ptr);
    }
    delete(p);
    L = nullptr;
    return true;
}

typedef struct LNode {
    ElemType data;
    struct LNode* next;
}LNode, *LinkList;

//传入一个链表指针，生成一个带有空的头结点的单链表
bool InitList(LinkList& L){
    LNode* head = new(LNode);
    if (!head)
        return false;
    L = head;
    head->next = nullptr;
    head->data = NULL;
    return true;
}
//求表长
int Length(LinkList L) {
    int num_node = 0;
    LNode* p = L;
    while (p->next) {
        p = p->next;
        num_node++;
    }
    return num_node;
}

//按值查找操作。在表L中查找具有给定关键字值的元素
int LocateElem(LinkList L, ElemType e) {
    LNode* p = L;
    int index = 0;
    while (p->next) {
        p = p->next;
        index++;
        if (p->data == e)
            return index;
    }
    return -1;
}

//按位查找操作。获取表L中第i个位置的元素的值。
ElemType GetElem(LinkList L, int i) {
    if (i<1 || i > Length(L))
        return NULL;

    LNode* p = L;
    for (int j = 0; j < i; j++)
